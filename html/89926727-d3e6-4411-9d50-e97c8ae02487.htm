DataTable을 사용하여 간단한 그리드 컨트롤을 만드는 방법을 제공합니다.

간단한 그리드 컨트롤 만들기

  1. 새로운 프로젝트를 만든다.

    Note

    새로운 프로젝트를 만든후 추가적으로 폼의 사이즈는 500, 500 이고 그리드 컨트롤의 Dock속성을 Fill로 설정하였습니다.

  2. DataTable 의 인스턴스를 생성한다.

    CopyC#
    DataTable dataTable = new DataTable("DataTable");

  3. 데이터 타입별로 DataColumn의 인스턴스를 생성한다.

    CopyC#
    dataTable.Columns.Add("Int32", typeof(int));
dataTable.Columns.Add("Float", typeof(float));
dataTable.Columns.Add("Boolean", typeof(bool));
dataTable.Columns.Add("Color", typeof(Color));

  4. DataSource()()()()에 설정한다.

    CopyC#
    this.gridControl1.DataSource = dataTable;

Example

CopyC#
using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Windows.Forms;

namespace SimpleGridControl
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();

            DataTable dataTable = new DataTable("DataTable");
            dataTable.Columns.Add("Int32", typeof(int));
            dataTable.Columns.Add("Float", typeof(float));
            dataTable.Columns.Add("Boolean", typeof(bool));
            dataTable.Columns.Add("Color", typeof(Color));

            for (int i = 0; i < 10; i++)
            {
                dataTable.Rows.Add();
            }

            this.gridControl1.DataSource = dataTable;
        }
    }
}
